85  static JMATH_API bool IsPowerOfTwo(const unsigned int num)
86  {
87  unsigned int n = num;
88  //shift bits to right (divide by two) until get a set bit that is most significant or not
89  while (((n & 1) == 0) && n > 1) //while num is even and greater than 1;
90  {
91  n >>= 1; //shift bits to the right one space
92  }
93  //if found a bit set to 1 and it's not most significant (i.e. n > 1) then return false
94  //else return true;
95  return (n == 1);
96  }

101  static JMATH_API unsigned int GetNextPowerOfTwo(const unsigned int value)
102  {
103  unsigned int v = value;
104  //check edge case of 0
105  if (value == 0)
106  return 0;
107  /*copy the most significant bit to all lower bits, then add 1 which results in carry that
108  sets all lower bits to 0 and one bit beyond to 1. If the original number was a power of 2
109  then the decrement will reduce it by one less, so then round up to original value.
110  */
111  v--;
112  v |= v >> 1;
113  v |= v >> 2;
114  v |= v >> 4;
115  v |= v >> 8;
116  v |= v >> 16;
117  v++;
118 
119  return v;
120  }
